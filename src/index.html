import math

a = Point(3, 4)
b = Point(0, 2)
c = Point(5, 1)
k1 = Line(a, b)
k2 = Line(b, c)
k3 = Line(c, a)

@a.when_moved
def find_area():
    ab = Distance(a, b)
    bc = Distance(b, c)
    ca = Distance(c, a)
    s = 0.5 * (ab + bc + ca)
    A = math.sqrt(s * (s - ab) * (s - bc) * (s - ca))
    print(A)
